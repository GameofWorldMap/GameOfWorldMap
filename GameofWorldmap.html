<head>

    <meta charset="UTF-8">

    <title>Khám phá trò chơi dân gian thế giới</title>

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css" />

    <style>

        #map { height: 600px; width: 100%; }

        .info-box {

            position: absolute;

            top: 10px;

            right: 10px;

            background: white;

            padding: 10px;

            border: 1px solid #ccc;

            display: none;

        }

    </style>

</head>

<body>

    <div id="map"></div>

    <div id="info" class="info-box"></div>

    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>

    <script>

        // Khởi tạo bản đồ

        const map = L.map('map', {

            zoomControl: true,

            minZoom: 2,

            maxZoom: 10

        }).setView([20, 0], 2);

        // Thêm lớp bản đồ từ OpenStreetMap

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'

        }).addTo(map);

        // Dữ liệu mẫu về quốc gia và trò chơi dân gian (có thể mở rộng)

        const countriesData = {

            "Vietnam": {

                color: "#FF5733",

                game: "Trò chơi Rồng Rắn Lên Mây: Trẻ em xếp hàng dài, một người làm 'thầy thuốc', người dẫn đầu hát và đối đáp để bảo vệ 'đuôi' khỏi bị bắt."

            },

            "Japan": {

                color: "#33FF57",

                game: "Trò chơi Daruma-san ga Koronda: Tương tự 'Tượng gỗ', trẻ em tiến lên khi người dẫn quay lưng và dừng khi người dẫn quay lại."

            },

            "Brazil": {

                color: "#FFFF33",

                game: "Trò chơi Queimada: Hai đội ném bóng để loại đối thủ, ai bị bóng chạm phải ra ngoài."

            }

            // Thêm quốc gia khác nếu muốn

        };

        // Thêm lớp GeoJSON (ranh giới quốc gia)

        fetch('https://raw.githubusercontent.com/datasets/geo-countries/master/data/countries.geojson')

            .then(response => response.json())

            .then(data => {

                L.geoJSON(data, {

                    style: function(feature) {

                        const countryName = feature.properties.ADMIN;

                        const countryInfo = countriesData[countryName] || { color: "#cccccc" };

                        return {

                            fillColor: countryInfo.color,

                            weight: 1,

                            opacity: 1,

                            color: 'white',

                            fillOpacity: 0.7

                        };

                    },

                    onEachFeature: function(feature, layer) {

                        const countryName = feature.properties.ADMIN;

                        layer.on({

                            click: function(e) {

                                const infoBox = document.getElementById('info');

                                const countryInfo = countriesData[countryName];

                                if (countryInfo) {

                                    infoBox.style.display = 'block';

                                    infoBox.innerHTML = `

                                        <h3>${countryName}</h3>

                                        <p>${countryInfo.game}</p>

                                    `;

                                } else {

                                    infoBox.style.display = 'block';

                                    infoBox.innerHTML = `<p>Chưa có thông tin trò chơi cho ${countryName}.</p>`;

                                }

                                map.flyToBounds(e.target.getBounds());

                            }

                        });

                    }

                }).addTo(map);

            });

        // Đóng hộp thông tin khi nhấp ra ngoài

        map.on('click', function(e) {

            document.getElementById('info').style.display = 'none';

        });

    </script>

</body>

</html>
